#include <bits/stdc++.h>
using namespace std;
const int N = 1002;
double l, r,a[N];
int n;
double f(double x) {
	double ans = 0;
	for (int i = 1; i <= n; i++) {
		ans = x * ans + a[i];
	}
	return ans;
}

int main() {
	cin >> n >> l >> r;
	n++;
	for (int i = 1; i <= n; i++) {
		cin >> a[i];
	}

	while (fabs(r - l) > 1e-6) {
		double rr = (2 * r + l) / 3;
		double ll = (r + 2 * l) / 3;
		if (f(rr) > f(ll)) {
			l = ll;
		}
		else {
			r = rr;
		}
	}
	cout << l << endl;
}
