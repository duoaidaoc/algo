using namespace std;

int n, l, s;
set<pair<int, int>>nodes;
int mode[60][60];
int main() {
	cin >> n >> l >> s;
	for (int i = 1; i <= n; i++) {
		int x, y;
		cin >> x >> y;
		nodes.insert({ x,y });
	}
	for (int i = 0; i <= s; i++) {
		for (int j = 0; j <= s; j++) {
			cin >> mode[s - i][j];
		}
	}
	int cnt = 0;
	for (auto& node : nodes) {
		int st = 0;
		for (int i = 0; i <= s; i++) {
			for (int j = 0; j <= s; j++) {
				int xx = node.first + i, yy = node.second + j;
				int cu = nodes.count({ xx,yy });
				if ((mode[i][j]&& !cu)||(!mode[i][j] && cu)|| xx > l || yy > l) {
					st = 1;
					break;
				}
			}
		}
		if (!st)cnt++;
	}
	cout << cnt;
}
