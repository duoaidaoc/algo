#include <bits/stdc++.h>
#define ll long long
using namespace std;
const int N = 1e6 + 20;
int n, k;
ll sum[N] = { 0 };
int main() {
	cin >> n >> k;
	string s;
	cin >> s;
	for (int i = 0; i < n; i++) {
		sum[i + 1] = sum[i] + s[i] - '0';
	}
	int i = 0;
	int j = 1;
	int len = n - k;
	while (i + len - 1 <= n - 1 && j + len - 1 <= n - 1) {
		int tmp = 0;
		while (s[i + tmp] == s[j + tmp] && max(i + tmp, j + tmp) < n - 1)
			tmp++;
		if (s[i + tmp] < s[j + tmp]) {
			if (i < j && i + tmp + 1 > n - len) {
				i = n - len;
				j = n + 1;
				break;
			}
			i = i + tmp + 1;
		}
		else {
			if (j < i && j + tmp + 1 > n - len) {
				j = n - len;
				i = n + 1;
				break;
			}
			j = j + tmp + 1;
			
		}
		if (i == j)i++;
	}
	ll ans = min(i, j);
	ll addd = sum[n] - sum[ans + len] + sum[ans];
	ll ext = 0;
	stack<char>st;
	for (int low = ans + len - 1; low >= ans; low--) {
		char thi = (s[low] - '0' + addd % 10 + ext) % 10 + '0';
		ext = (s[low] - '0' + addd % 10 + ext) / 10;
		addd /= 10;
		st.push(thi);
	}

	while (addd != 0) {
		st.push((addd + ext) % 10 + '0');
		ext = (addd % 10 + ext) / 10;
		addd /= 10;
	}

	if (ext != 0) {
		st.push(ext + '0');
	}

	while (!st.empty()) {
		putchar(st.top());
		st.pop();
	}
}
